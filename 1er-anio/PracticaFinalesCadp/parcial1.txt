procedure insertar(var l:lista; v: venta);
var
	ant, act, nue: lista;
begin
	new(nue);
	nue^.dato := v;
	nue^.sig := nil;
	if (l = nil) then
		l := nue
	else begin
		act := l;
		while (act <> nil) and (act^.dato.tipopago < v.tipopago) do begin
			ant := act;
			act := act^.sig;
		end;
		if (act = l) then begin
			nue^.sig := l;
			l := nue;
		end
		else
			ant^.sig := nue;
		nue^.sig := act;
	end;
		
end;
function cumple (num : integer): boolean;
var
	dig,par,imp: integer;
begin
	par :=0; imp := 0;
	while (num <> 0) do begin
		dig ::= num mod 10;
		if (dig mod 2 = 0) then 
			par := par + 1
		else
			imp := imp + 1;
	end;
	cumple := (par > imp);
end;
